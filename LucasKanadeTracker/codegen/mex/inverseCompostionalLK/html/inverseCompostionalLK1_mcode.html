1function opticalFlow = inverseCompostionalLK(previousFrame, currentFrame, ...
 2    pointsToTrack, windowRadiousY, windowRadiousX, maxIterations, ...
 3    stopThreshold, weightingKernel, initialOpticalFlow)
 4    % References
 5    % ----------
 6    % 1. Baker, S. & Matthews, I. Lucas-Kanade 20 Years On: A Unifying 
 7    % Framework International Journal of Computer Vision, 2004, 56, 221-255
 8
 9    % Compute spatial gradients of previous frame
10    [gradientX, gradientY] = gradient(previousFrame);
11    
12    % Initialize optical flow vectors
13    opticalFlow = zeros(size(pointsToTrack));
14    
15    % Iterate for every given point to track
16    for currentPointIdx = 1:size(pointsToTrack,1)
17    
18        % Initialize optical flow estimation for the current point
19        currentFlow = [0; 0];
20        
21        % Determine current point tracking window 
22        windowYRangeMin = pointsToTrack(currentPointIdx, 2) - windowRadiousY;
23        windowYRangeMax = pointsToTrack(currentPointIdx, 2) + windowRadiousY;
24        windowXRangeMin = pointsToTrack(currentPointIdx, 1) - windowRadiousX;
25        windowXRangeMax = pointsToTrack(currentPointIdx, 1) + windowRadiousX;
26     
27        [windowPixelsX, windowPixelsY] = meshgrid(windowXRangeMin : windowXRangeMax, windowYRangeMin : windowYRangeMax); 
28        
29        % Get spatial gradient for current window
30        gradientYWindow = bilinearInterpolate(gradientY, windowPixelsY, windowPixelsX);
31        gradientXWindow = bilinearInterpolate(gradientX, windowPixelsY, windowPixelsX);
32        
33        % Get template for current window
34        templateWindow = bilinearInterpolate(previousFrame, windowPixelsY, windowPixelsX);
35        
36        % Compute Hessian matrix
37        gradientMomentXX = sum(sum(gradientXWindow .* gradientXWindow .* weightingKernel));
38        gradientMomentXY = sum(sum(gradientXWindow .* gradientYWindow .* weightingKernel));
39        gradientMomentYY = sum(sum(gradientYWindow .* gradientYWindow .* weightingKernel));
40        hessian = [gradientMomentXX gradientMomentXY; gradientMomentXY gradientMomentYY];
41        
42        % Estimate and refine optical flow for current point
43        for i = 1:maxIterations
44
45            % Warp current frame window with last opitcal flow estimation
46            windowWarped = bilinearInterpolate(currentFrame, ...
47                   windowPixelsY + currentFlow(2) + initialOpticalFlow(currentPointIdx, 2), ...
48                   windowPixelsX + currentFlow(1) + initialOpticalFlow(currentPointIdx, 1));
49
50            % Compute error image for current window
51            errorImage = windowWarped - templateWindow;
52            
53            % Update steppest descent parameters
54            steppestDescentUpdate = [sum(sum(gradientXWindow .* errorImage .* weightingKernel)); ... 
55                    sum(sum(gradientYWindow .* errorImage .* weightingKernel))];
56
57            % Compute optical flow change
58            flowChange = hessian \ steppestDescentUpdate;
59
60            % Update optical flow by composition with warp inversion
61            currentFlow = currentFlow - flowChange;
62
63            % Check stop condition
64            if norm(flowChange) < stopThreshold
65                
66                % Stop if converged
67                break;
68
69            end
70
71        end
72
73        % Save result for current point
74        opticalFlow(currentPointIdx, :) = currentFlow;
75        
76    end
77    
78end
